Hi

I tried the following sample and it works.

<zk>
<style>
.button {
	color: red;
	font-size: 20px;
}
</style>
<vbox id="vbox" apply="org.zkoss.bind.BindComposer" viewModel="@id('vm') @init('forum.ButtonVM')">
	<button id="cmd" label="cmd" onClick="@command('cmd')" sclass="@load(vm.sclass, after='cmd')" mold="os"/>
</vbox>
</zk>


public class ButtonVM {
	private String sclass = "button";

	public String getSclass() {
		return sclass;
	}

	@NotifyChange
	public void setSclass(String sclass) {
		this.sclass = sclass;
	}
	
	@Command
	public void cmd() {
	}
}

The problem seems that your button sclass is loaded from other View Model and it cannot recognize it.

package forum;

import org.zkoss.bind.annotation.Command;
import org.zkoss.bind.annotation.NotifyChange;

public class ButtonVM {
	private String sclass = "button";
	private boolean btnDisabled;

	public String getSclass() {
		return sclass;
	}


	public void setBtnDisabled(boolean btnDisabled) {
		this.btnDisabled = btnDisabled;
	}

	public boolean isBtnDisabled() {
		return btnDisabled;
	}
	@NotifyChange
	public void setSclass(String sclass) {
		this.sclass = sclass;
	}
	
	@Command
	@NotifyChange("btnDisabled")
	public void cmd() {
		btnDisabled=true;
	}
	
}
